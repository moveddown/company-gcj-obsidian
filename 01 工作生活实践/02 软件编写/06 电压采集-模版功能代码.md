> [!note] 电压采集类型
> 1. 直接内部电压采集：不占用外部 ADC 引脚，采内部 1.2V 基准，反推 MCU 当前供电电压 VDD。
> 2. 外部电压分压采集 + 内部电压校准：占用 1 个外部 ADC 引脚，同时采外部分压点和内部 1.2V 基准，用于还原外部电池/高压输入。
> 3. 分压使能脚控制导通 + 外部分压采集 + 内部电压校准：占用 1 个 ADC 引脚和 1 个开关控制引脚，适合极低功耗场景。本页暂不展开。

> [!info] 基础理论
> ADC 输出的不是电压绝对值，而是输入电压相对参考电压的比例。更完整的理论可以看：[[02 理论支持和基础/05 外设理论知识/01 电压采集ADC-单片机外设知识|ADC采集电压]]。

# 06 电压采集-模版功能代码

## 基础公式速记

ADC 的本质公式：

```text
ADC = Vin / Vref * ADC_MAX
```

12 位 ADC 时：

```text
ADC_MAX = 4095
```

普通外部采样，参考电压固定：

```text
Vin = ADC * Vref / 4095
```

采内部 1.2V 基准，反推 MCU 供电电压：

```text
ADC_bgr = Vbgr / VDD * 4095
VDD = Vbgr * 4095 / ADC_bgr
```

外部分压还原：

```text
Vpin = Vin * Rbottom / (Rtop + Rbottom)
Vin = Vpin * (Rtop + Rbottom) / Rbottom
```

外部分压 + 内部基准校准时：

```text
ADC_div = Vpin / Vref * 4095
ADC_bgr = Vbgr / Vref * 4095
Vpin = ADC_div * Vbgr / ADC_bgr
Vin = ADC_div * Vbgr / ADC_bgr * (Rtop + Rbottom) / Rbottom
```

其中：

```text
ADC_div：外部分压点 ADC 原始值
ADC_bgr：内部 1.2V 基准 ADC 原始值
Vbgr：内部基准电压，常用 1200mV
Vpin：ADC 引脚上的分压后电压
Vin：待测电池/外部输入真实电压
```

## 方案 A：直接内部电压采集，无外部引脚

### 适用场景

电池直接给 MCU 供电，需要测 MCU 当前供电电压 VDD。此时 ADC 参考电压通常就是 VDD，如果直接采 VDD 会一直接近满量程，测不出电池变化。

正确做法是采内部固定 1.2V 基准，然后反推 VDD：

```text
VDD = 1200mV * 4095 / ADC_bgr
```

```c
#include "bsp_lvd.h"
#include "cw32l010.h"
#include "cw32l010_adc.h"
#include "cw32l010_sysctrl.h"

/* 如果芯片提供内部基准校准值，可改成读校准区；没有校准值就用 1200mV。 */
static uint32_t lvd_get_bgr_mv(void)
{
    return LVD_INTERNAL_VREF_MV;
}

static uint32_t lvd_calc_vdd_mv(uint16_t adc_bgr_raw)
{
    uint32_t bgr_mv;

    if (adc_bgr_raw == 0U)
    {
        return 0U;
    }

    bgr_mv = lvd_get_bgr_mv();
    return (LVD_ADC_MAX_VALUE * bgr_mv) / adc_bgr_raw;
}

static void bsp_adc_init(void)
{
    ADC_InitTypeDef ADC_InitStructure = {0};

    __SYSCTRL_ADC_CLK_ENABLE();
    ADC_DeInit();

    ADC_InitStructure.ADC_ClkDiv = ADC_Clk_Div8;
    ADC_InitStructure.ADC_ConvertMode = ADC_ConvertMode_Once;
    ADC_InitStructure.ADC_SQREns = ADC_SqrEns0to0;
    ADC_InitStructure.ADC_IN0.ADC_InputChannel = ADC_InputVref1P2;
    ADC_InitStructure.ADC_IN0.ADC_SampTime = ADC_SampTime390Clk;
    ADC_Init(&ADC_InitStructure);

    CW_ADC->CR_f.BGREN = 1;
    ADC_ClearITPendingAll();
    ADC_Enable();
}

static void bsp_adc_deinit(void)
{
    CW_ADC->CR_f.BGREN = 0;
    ADC_Disable();
    __SYSCTRL_ADC_CLK_DISABLE();
}

static uint8_t lvd_calc_bat_level(uint32_t bat_mv)
{
    if (bat_mv <= LVD_BAT_LEVEL_MIN_MV)
    {
        return 0U;
    }

    if (bat_mv >= LVD_BAT_LEVEL_MAX_MV)
    {
        return 100U;
    }

    return (uint8_t)((bat_mv - LVD_BAT_LEVEL_MIN_MV) * 100U /
                     (LVD_BAT_LEVEL_MAX_MV - LVD_BAT_LEVEL_MIN_MV));
}

{
    static uint8_t low_bat_active = 0U;

    {
        if (low_bat_active == 0U)
        {
            low_bat_active = 1U;

        }
    }
    else if (bat_mv > LVD_MAX_VAL)
    {
        if (low_bat_active != 0U)
        {
            low_bat_active = 0U;
            /* TODO: 清除低电标志、触发低电解除灯效 */
        }
    }

    bat_level = lvd_calc_bat_level(bat_mv);
    if (bat_level < 2U)
    {
        bat_level = 2U;
    }

    /* TODO: 保存 bat_mv 和 bat_level 到业务状态结构 */

uint32_t lvd_sample_vdd_mv_blocking(void)
{
    uint32_t adc_sum = 0U;
    uint16_t adc_avg;
    uint32_t vdd_mv;
    uint8_t i;

    bsp_adc_init();
    /* TODO: 等待 LVD_STARTUP_WAIT_MS，让内部基准稳定。 */

    for (i = 0U; i < LVD_MAX_ADC_BUF; i++)
    {
        ADC_ClearITPendingBit(ADC_IT_EOC);
        ADC_SoftwareStartConvCmd(ENABLE);
        while (!(CW_ADC->ISR & ADC_ISR_EOC_Msk))
        {
        }
        ADC_ClearITPendingBit(ADC_IT_EOC);
        adc_sum += ADC_GetConversionValue(ADC_RESULT_0);
    }

    adc_avg = (uint16_t)(adc_sum / LVD_MAX_ADC_BUF);
    vdd_mv = lvd_calc_vdd_mv(adc_avg);
    lvd_update_state(vdd_mv);

    bsp_adc_deinit();
    return vdd_mv;
}
```

### 移植要点

- 只需要内部 ADC 通道，不占用外部引脚。
- 如果芯片有内部 1.2V 校准值，优先用校准值替代固定 `1200UL`。
- 低功耗场景下，采样结束后关闭内部基准、ADC 和 ADC 时钟。
- 低电判断需要回差：例如低于 3.5V 进入低电，高于 3.8V 才解除。

## 方案 B：外部分压 + 内部基准校准，一个 ADC 引脚

### 适用场景

用于测外部电池、高压输入、充电输入、太阳能输入等。待测电压可能高于 ADC 可承受范围，所以先用电阻分压，再由 ADC 采分压点。

同时采内部 1.2V 基准，是为了校准 ADC 当前参考电压。这样即使 MCU 的 VDD 不是固定 3.3V，也能计算外部真实电压。

这套方案里，两个 ADC 值来自同一个 ADC，只是一次序列采两个通道：

```text
ADC_div：外部分压点通道的 ADC 原始值
ADC_bgr：内部 1.2V 基准通道的 ADC 原始值
```

### 换算函数模板

```c
typedef struct
{
    uint16_t adc_div_raw;  // 外部分压点 ADC 平均值
    uint16_t adc_bgr_raw;  // 内部 1.2V ADC 平均值
    uint16_t adc_pin_mv;   // ADC 引脚实际电压，单位 mV
    uint16_t bat_mv;       // 分压还原后的真实电压，单位 mV
} lvd_sample_result_t;

static uint32_t lvd_calc_adc_pin_mv(uint32_t adc_div_raw, uint32_t adc_bgr_raw)
{
    if (adc_bgr_raw == 0U)
    {
        return 0U;
    }

    return (adc_div_raw * LVD_INTERNAL_VREF_MV) / adc_bgr_raw;
}

static uint32_t lvd_calc_bat_mv(uint32_t adc_div_raw, uint32_t adc_bgr_raw)
{
    uint32_t adc_pin_mv;

    if ((adc_bgr_raw == 0U) || (LVD_BAT_DIVIDER_RATIO_DEN == 0U))
    {
        return 0U;
    }

    adc_pin_mv = lvd_calc_adc_pin_mv(adc_div_raw, adc_bgr_raw);
    return (adc_pin_mv * LVD_BAT_DIVIDER_RATIO_NUM) / LVD_BAT_DIVIDER_RATIO_DEN;
}

static uint8_t lvd_calc_bat_level(uint32_t bat_mv)
{
    if (bat_mv <= LVD_BAT_LEVEL_MIN_MV)
    {
        return 0U;
    }

    if (bat_mv >= LVD_BAT_LEVEL_MAX_MV)
    {
        return 100U;
    }

    return (uint8_t)((bat_mv - LVD_BAT_LEVEL_MIN_MV) * 100U /
                     (LVD_BAT_LEVEL_MAX_MV - LVD_BAT_LEVEL_MIN_MV));
}
```

```c
static int lvd_sample_external_voltage(lvd_sample_result_t *result)
{
    uint32_t adc_div_sum = 0U;
    uint32_t adc_bgr_sum = 0U;
    uint32_t adc_div_avg;
    uint32_t adc_bgr_avg;
    uint8_t i;

    if (result == 0)
    {
        return -1;
    }

    bsp_lvd_gpio_init();
    bsp_adc_init();
    /* TODO: 等待 LVD_ADC_STABLE_TIME_MS，让分压点和内部基准稳定。 */

    for (i = 0U; i < LVD_MAX_ADC_BUF; i++)
    {
        ADC_ClearITPendingBit(ADC_IT_EOS);
        ADC_SoftwareStartConvCmd(ENABLE);
        while (!(CW_ADC->ISR & ADC_ISR_EOS_Msk))
        {
        }
        ADC_ClearITPendingBit(ADC_IT_EOS);

        adc_div_sum += ADC_GetConversionValue(ADC_RESULT_0);
        adc_bgr_sum += ADC_GetConversionValue(ADC_RESULT_1);
    }

    adc_div_avg = adc_div_sum / LVD_MAX_ADC_BUF;
    adc_bgr_avg = adc_bgr_sum / LVD_MAX_ADC_BUF;

    result->adc_div_raw = (uint16_t)adc_div_avg;
    result->adc_bgr_raw = (uint16_t)adc_bgr_avg;
    result->adc_pin_mv = (uint16_t)lvd_calc_adc_pin_mv(adc_div_avg, adc_bgr_avg);
    result->bat_mv = (uint16_t)lvd_calc_bat_mv(adc_div_avg, adc_bgr_avg);

    bsp_adc_deinit();
    return (adc_bgr_avg == 0U) ? -1 : 0;
}
```

### 分压倍率怎么填

硬件结构：

```text
Vin
 |
 Rtop
 |
 +---- ADC 引脚 Vpin
 |
 Rbottom
 |
GND
```

分压关系：

```text
Vpin = Vin * Rbottom / (Rtop + Rbottom)
Vin = Vpin * (Rtop + Rbottom) / Rbottom
```

所以：

```text
LVD_BAT_DIVIDER_RATIO_NUM / LVD_BAT_DIVIDER_RATIO_DEN = (Rtop + Rbottom) / Rbottom
```

例：真实电压是 ADC 引脚电压的 6.4 倍：

```c
#define LVD_BAT_DIVIDER_RATIO_NUM     640UL
#define LVD_BAT_DIVIDER_RATIO_DEN     100UL
```

## 移植检查清单

- [ ] 确认 ADC 是否为 12 位；如果不是，修改 `LVD_ADC_MAX_VALUE`。
- [ ] 确认内部基准是否为 1.2V；如果不是，修改 `LVD_INTERNAL_VREF_MV`。
- [ ] 确认内部基准是否需要单独使能，例如 `BGREN`。
- [ ] 确认内部基准、外部分压点是否需要等待稳定。
- [ ] 方案 B 确认外部 ADC 引脚和 ADC 通道是否对应。
- [ ] 方案 B 确认 ADC 引脚已配置为模拟输入。
- [ ] 方案 B 确认分压电阻比例，并换算成 `NUM / DEN`。
- [ ] 确认最大输入电压分压后不会超过 ADC 引脚允许电压。
- [ ] 确认低电阈值和恢复阈值有回差，避免电压抖动反复报警。
- [ ] 低功耗产品确认采样结束后关闭 ADC、ADC 时钟、内部基准。
- [ ] 如果有分压使能脚，采样前打开，采样后关闭；本页暂未展开该第三种模板。

## 快速选型

```text
只想测 MCU 自己的电池供电电压：
选方案 A，采内部 1.2V 反推 VDD。

要测外部高压、电池包、充电输入、太阳能输入：
选方案 B，外部分压 + 内部 1.2V 校准。

外部分压电阻长期耗电不可接受：
在方案 B 基础上增加分压使能脚，采样时才导通。
```
